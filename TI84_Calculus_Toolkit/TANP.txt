:X->P:Y->Q

:.001->H

:(A-B)/(2H)->U

:Q-H->Y:expr(Str1)->B
:(A-B)/(2H)->V
:ClrHome
:Disp "PLANE EQUATION:"
:Disp "Z =",Z
:Disp "+ (Fx)(X-P)"
:Disp "+ (Fy)(Y-Q)"
:Disp ""
:Disp "Fx =",U
:Disp "Fy =",V
:Disp "Z0 =",Z
:Disp "P =",P
:Disp "Q =",Q
